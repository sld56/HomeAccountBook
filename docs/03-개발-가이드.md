## 4. 상태 관리 패턴

### 4.1 Zustand 스토어 예시
```ts
// src/stores/transactionStore.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import type { Transaction } from '@/types/domain';
import { seedTransactions } from '@/data/transactions';

type TxState = {
  transactions: Transaction[];
  add: (tx: Omit<Transaction, 'id'>) => void;
  update: (id: string, patch: Partial<Transaction>) => void;
  remove: (id: string) => void;
};

export const useTransactions = create<TxState>()(
  persist(
    (set) => ({
      transactions: seedTransactions,
      add: (tx) => set((s) => ({
        transactions: [{ ...tx, id: crypto.randomUUID() }, ...s.transactions],
      })),
      update: (id, patch) => set((s) => ({
        transactions: s.transactions.map((t) => t.id === id ? { ...t, ...patch } : t),
      })),
      remove: (id) => set((s) => ({
        transactions: s.transactions.filter((t) => t.id !== id),
      })),
    }),
    { name: 'gagyebu-transactions' }
  )
);
```

## 5. 포맷팅 함수

```ts
// src/lib/format.ts
import type { CurrencyMode, TransactionKind } from '@/types/domain';

export const fmt = {
  money(n: number, mode: CurrencyMode = 'won'): string {
    const abs = Math.abs(Math.round(n));
    const formatted = abs.toLocaleString('ko-KR');
    const sign = n < 0 ? '-' : '';
    if (mode === 'symbol') return `${sign}₩${formatted}`;
    if (mode === 'korean') return `${sign}${fmt.korean(abs)}`;
    return `${sign}${formatted}원`;
  },

  korean(n: number): string {
    const eok = Math.floor(n / 100_000_000);
    const man = Math.floor((n % 100_000_000) / 10_000);
    const rem = n % 10_000;
    const parts: string[] = [];
    if (eok) parts.push(`${eok}억`);
    if (man) parts.push(`${man.toLocaleString()}만`);
    parts.push(`${rem.toLocaleString()}원`);
    return parts.join(' ');
  },

  signed(n: number, kind: TransactionKind, mode: CurrencyMode = 'won'): string {
    const s = kind === 'in' ? '+' : '−';
    return `${s}${fmt.money(n, mode)}`;
  },

  short(n: number): string {
    const abs = Math.abs(n);
    if (abs >= 100_000_000) return `${(abs / 100_000_000).toFixed(1)}억`;
    if (abs >= 10_000)     return `${Math.round(abs / 10_000)}만`;
    return abs.toLocaleString('ko-KR');
  },

  date(d: string | Date, style: 'short' | 'long' | 'day' = 'short'): string {
    const dt = typeof d === 'string' ? new Date(d) : d;
    const days = ['일', '월', '화', '수', '목', '금', '토'];
    if (style === 'long') return `${dt.getFullYear()}년 ${dt.getMonth()+1}월 ${dt.getDate()}일 ${days[dt.getDay()]}요일`;
    if (style === 'day')  return `${dt.getDate()}일 ${days[dt.getDay()]}요일`;
    return `${dt.getMonth() + 1}월 ${dt.getDate()}일`;
  },
};
```

## 6. 차트 컴포넌트 패턴

모든 차트는 다음 패턴을 따른다.

```tsx
type ChartProps = {
  data: SomeDataShape[];
  currencyMode?: CurrencyMode;
  // 차트별 옵션 ...
};

export function SomeChart({ data, currencyMode = 'won' }: ChartProps) {
  const [active, setActive] = useState<number | null>(null);
  const onMove = (e: React.PointerEvent) => { /* idx 계산 */ };
  return (
    <div style={{ position: 'relative' }} onPointerLeave={() => setActive(null)}>
      <svg onPointerMove={onMove} onPointerDown={onMove} style={{ touchAction: 'none' }}>
        {/* 트랙, 라인, 점 ... */}
      </svg>
      {active != null && (
        <ChartTooltip x={...} y={...}>
          {/* 콘텐츠 */}
        </ChartTooltip>
      )}
    </div>
  );
}
```

**공통 규칙**
- `touchAction: 'none'`을 SVG에 부여 (스크롤 충돌 방지)
- 호버 + 터치 모두 `onPointerEnter` / `onPointerDown`로 통합
- 툴팁 위치는 viewport 끝에서 자동 반전
- 비활성 상태 디폴트 표시 (총합 등) → 활성 시 상세

## 10. 접근성

- 모든 버튼은 `<button>` 또는 `role="button"`
- 폼 필드는 `<label htmlFor>` 필수
- 모달: `role="dialog" aria-modal="true" aria-labelledby="modal-title"`
- 차트는 `<svg role="img" aria-label="...">` + `<title>`/`<desc>`
- 색상만으로 정보 전달 금지 (수입/지출은 + / − 부호도 함께)
- 키보드 포커스 링: `:focus-visible { outline: 2px solid var(--coral); outline-offset: 2px; }`
